error_Code = -20000，        // 该函数的传入参数不正确(The incoming parameter of the function is incorrect)
error_Code = -20019，        // AV模块尚未初始化(AV module has not been initialized)
error_Code = -20002，        // AV通道数已达到最大值(The number of AV channels has reached the maximum)
error_Code = -20004，        // AV无法创建线程(AV cannot create thread)
error_Code = -20007，        // 指定的AV服务器没有响应(The specified AV server is not responding)
error_Code = -20015，        // 远程站点已关闭(The remote site is closed)
error_Code = -20016，        // 此会话已断开，因为远程站点在指定的超时到期后没有任何响应(This session has been disconnected because the remote site did not respond after the specified timeout expired)
error_Code = -20010，        // 指定AV通道的会话无效(The session of the specified AV channel is invalid)
error_Code = -20018，        // 客户端启动过程终止(Client startup process terminated)
error_Code = -20011，        // 某些操作期间指定的超时已到期(The timeout specified during some operations has expired)
error_Code = -20009，        // 由于查看帐户或密码不正确，客户端验证失败(Client verification failed due to incorrect account or password)
error_Code = -20023，        // UID是精简版UID(UID is a simplified UID)
error_Code = -10，           // 指定的UID未获得许可或已过期(The specified UID is not licensed or has expired)
error_Code = -90，           // 所有服务器响应都找不到设备(No device found by any server response)
